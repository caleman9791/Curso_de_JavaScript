Ejemplo

// Definimos la matriz.
let elementos = [ 'Elemento 1' , 'Elemento 2' ];

// Desplegar resultado en consola.
console . log ( elementos .length);

// Agregar el valor dentro del índice de
// la matriz a una variable de forma manual.
let primer_elemento = elementos [ 0 ];
let segundo_elemento = elementos [ 1 ];

// Desplegar resultado en consola.
console . log ( primer_elemento );
console . log ( segundo_elemento );

console . log ( "===============================" );

// Esto es un bucle y recorrerá y mostrará cada índice en consola.
for ( let i = 0 ; i < elementos .length; i ++ {
console . log ( elementos [ i ]);
}

console . log ( "===============================" );

// Con estos parámetros podremos acceder al último valor de
// la matriz que es la longitud de la matriz menos uno y sabrás
// que el primer valor de la matriz es cero.
console . log ( elementos [ elementos .length - 1 ]);
